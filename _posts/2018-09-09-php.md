        区分验证码类型
        0:getNumberCode()
        1:getCharCode()
        2:getNumCharCode()
    
    protected function getNumberCode()
        $str=join('',range(0,9))
        return substr(str_shuffle($str),0,$this->number);
    
    protected function getCharCode()
        $str=join('',range('a','z'));
        $str=$str.strtoupper($str);
        return substr(str_shuffle($str),0,$this->number);

        创建画布

    protected function fillBack()
        imagefill()

        mt_rand()
